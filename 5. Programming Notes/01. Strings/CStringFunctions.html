C String Functions

strlen()

        #include <stdio.h>
        #include <string.h>
        
        int main()
        {
            char *alphabet = "abcdefghijklmnopqrstuvwxyz";
        
            printf("%d\n", sizeof(alphabet)); // 8
            printf("%d\n", strlen(alphabet)); // 26
        
            char Alphabet[50] = "abcdefghijklmnopqrstuvwxyz";
        
            printf("%d\n", sizeof(Alphabet)); // 50
            printf("%d\n", strlen(Alphabet)); // 26
        }

        }

Example 2:

        #include <stdio.h>
        #include <string.h>
        
        int main()
        {
            char str1[20] = "Karan ";
            char *str2 = "Suthar";
        
            char *str3 = strcat(str1, str2);
        
            printf("%d\n", sizeof(str1)); // 20
            printf("%d\n", strlen(str1)); // 12
            printf("%s\n", str1);         // Karan Suthar
        
            printf("%d\n", sizeof(str3)); // 8
            printf("%d\n", strlen(str3)); // 12
            printf("%s\n", str3);         // Karan Suthar
        }

strcpy()

        #include <stdio.h>
        #include <string.h>
        
        int main()
        {
            char *str1 = "Hello Jii, Kese Hain Aap";
            char str2[100];
        
            char *str3 = strcpy(str2, str1);
        
            printf("%d\n", sizeof(str1)); // 8
            printf("%d\n", strlen(str1)); // 24
            printf("%s\n", str1);         // Hello Jii, Kese Hain Aap
        
            printf("%d\n", sizeof(str2)); // 100
            printf("%d\n", strlen(str2)); // 24
            printf("%s\n", str2);         // Hello Jii, Kese Hain Aap
        
            printf("%d\n", sizeof(str3)); // 8
            printf("%d\n", strlen(str3)); // 24
            printf("%s\n", str3);         // Hello Jii, Kese Hain Aap
        }

strcmp()

        #include <stdio.h>
        #include <string.h>
        
        void main() 
        {
            char *str1 = "Hello";
            char str2[] = "Hello";
            char str3[] = "Hi"; 
            char str4[10] = "Karan";
            char str5[13] = "KaraN";
        
            printf("%d\n", strcmp(str1, str2)); // 0
            printf("%d\n", strcmp(str1, str3)); // -4
            printf("%d\n", strcmp(str4, str5)); // 32
            printf("%d\n", strcmp(str5, str4)); // -32
        }

strrev()

        #include <stdio.h>
        #include <string.h>
        
        int main()
        {
            char str[50] = "Karan Suthar";
            // char *str = "Karan Suthar";
        
            printf("%s\n", str);          // Karan Suthar 
            printf("%s\n", strrev(str));  // rahtuS naraK
        
            return 0;
        }

// Note: In C language we can use the function strrev() present in the header file string.h to reverse any given string, strrev() is a non-standard string library function ie. it is not standardly available with all the compilers, only some specific compilers support it, but we can write our own implementation of a function to a reverse a string, if the function strrev() is not available.

strchr()

        #include <stdio.h>
        #include <string.h>
        
        int main()
        {
            char str[50] = "Karan Suthar";
            // char *str = "Karan Suthar";
        
            char *ptr = strchr(str, 'r');
        
            printf("%s\n", str);          // Karan Suthar
            printf("%c\n", *ptr);         // r
            printf("%c\n", *(ptr + 1));   // a
            printf("%c\n", *(ptr + 2));   // n
            printf("%s\n", ptr);          // ran Suthar
        
            return 0;
        }
