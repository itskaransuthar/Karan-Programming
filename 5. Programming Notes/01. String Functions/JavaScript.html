JavaScript String Methods

length

  1. The length property returns the length of a string

Example:

        let str1 = "Karan"

        let len = str1.length
        console.log(len)          // 5

        let str2 = " Karan "
        console.log(str2.length)  // 7

at()

  1. The at() method returns the character at a specified index (position) in a string
  2. It allows the use of negative indexes while charAt() do not
  3. Now you can use str.at(-2) instead of str.charAt(str.length-2)

charAt()

  1. The charAt() method returns the character at a specified index (position) in a string

charCodeAt()

  1. The charCodeAt() method returns the code of the character at a specified index in a string

Example:

        str = "Karan Suthar"

        console.log(str[2])                      // r  
        console.log(str.at(2))                   // r              
        console.log(str.charAt(2))               // r                  
        console.log(str.charCodeAt(2))           // 114                           
        
        console.log(str[-2])                     // undefined            
        console.log(str.at(-2))                  // a       
        console.log(str.charAt(str.length - 2))  // a    
        console.log(str.charAt(-2))              //                    
        console.log(str.charCodeAt(-2))          // NaN                       
            
        console.log(str[100])                    // undefined             
        console.log(str.at(100))                 // undefined                
        console.log(str.charAt(100))             //                     
        console.log(str.charCodeAt(100))         // NaN

String Concatenation

  1. concat()
    concat(str1)
    concat(str1, str2)
    concat(str1, str2, ... strN)
  2. A new string containing the combined text of the strings provided

Example:

        let str1 = "Karan"
        let str2 = 'Suthar'
        
        let newStr1 = str1.concat(" ").concat(str2);
        let newStr2 = str1 + " " + str2
        let newStr3 = str1.concat(" ", str2)
        let newStr4 = str1.concat(" ", str2, " Jii", ", ", "Kese Hain Aap", "!!")
        
        console.log(str1)          // Karan                
        console.log(str2)          // Suthar                   
        console.log(newStr1)       // Karan Suthar                      
        console.log(newStr2)       // Karan Suthar                      
        console.log(newStr3)       // Karan Suthar              
        console.log(newStr4)       // Karan Suthar Jii, Kese Hain Aap!! 
        
        const hello = "Hello, ";
        console.log(hello.concat("Karan", ". Have a nice day!")); // Hello, Karan. Have a nice day! 
        
        const greetList = ["Hello", " ", "Aniket", "!"];

        "".concat(...greetList);   // "Hello Aniket!"         
        "".concat({});             // "[object Object]"           
        "".concat([]);             // ""          
        "".concat(null);           // "null"           
        "".concat(true);           // "true"           
        "".concat(4, 5);           // "45"

String Comparison

Example:

        const str1 = "Karan"
        const str2 = "KaraN"
        
        console.log(str1 == str2) // false
        console.log(str1 < str2)  // false
        console.log(str1 > str2)  // true

indexOf()

  1. indexOf(searchString)
    indexOf(searchString, position)
  2. The index of the first occurrence of searchString found, or -1 if not found

Example:

        const str = "Karan Suthar"
        
        let index1 = str.indexOf('r')
        let index2 = str.indexOf("r")
        let index3 = str.indexOf(" S")
        let index4 = str.indexOf(" s")
        let index5 = str.indexOf("ran")
        
        let index6 = str.indexOf('r', 2)
        let index7 = str.indexOf('r', 3)
        let index8 = str.indexOf(' S', 6)
        let index9 = str.indexOf(' s', 1)
        let index10 = str.indexOf("ran", 5)
        
        console.log(index1) // 2
        console.log(index2) // 2
        console.log(index3) // 5
        console.log(index4) // -1
        console.log(index5) // 2
        console.log(index6) // 2
        console.log(index7) // 11
        console.log(index8) // -1
        console.log(index9) // -1

replace()

  1. replace(pattern, replacement)
  2. A new string, with one, some, or all matches of the pattern replaced by the specified replacement

replaceAll()

  1. replaceAll(pattern, replacement)
  2. A new string, with all matches of a pattern replaced by a replacement

Example:

        str = "Karan Suthar"

        console.log(str.replace('a', 'i'));     // Kiran Suthar 
        console.log(str.replaceAll('a', 'i'));  // Kirin Suthir  
        console.log(str);                       // Karan Suthar
                                               
        console.log("abc".replace("", "-->"))   // -->abc

slice()

  1. slice(indexStart)
    slice(indexStart, indexEnd)
  2. A new string, with one, some, or all matches of the pattern replaced by the specified replacement

substring()

  1. substring(indexStart)
    substring(indexStart, indexEnd)
  2. substring() is similar to slice(), the difference is that start and end values less than 0 are treated as 0 in substring()

substr()

  1. substr(start)
    substr(start, length)
  2. substr() is similar to slice(), the difference is that the second parameter specifies the length of the extracted part

Example:

        str = "Karan Suthar"
        
        console.log(str.slice(2, 5));               // ran                
        console.log(str.slice(-4, -1));             // tha     
        console.log(str.slice(-4, 2));              //          
        console.log(str.slice(6));                  // Suthar       
        console.log(str.slice(-6));                 // Suthar        
        console.log(str.slice());                   // Karan Suthar      
        console.log(str.slice(5445));               //           
        console.log(str.slice(-445));               // Karan Suthar          
        console.log(str.slice(-31, 65));            // Karan Suthar             
        console.log(str.slice(30, 50));             //    
        
        console.log(str.substring(2, 5));           // ran                           
        console.log(str.substring(-4, -1));         //      
        console.log(str.substring(-4, 2));          // Ka                             
        console.log(str.substring(6));              // Suthar                        
        console.log(str.substring(-6));             // Karan Suthar                         
        console.log(str.substring());               // Karan Suthar                       
        console.log(str.substring(5445));           //                            
        console.log(str.substring(-445));           // Karan Suthar                           
        console.log(str.substring(-31, 65));        // Karan Suthar                              
        console.log(str.substring(30, 50));         //                              
        
        console.log(str.substr(2, 5));              // ran S                      
        console.log(str.substr(-4, -1));            //       
        console.log(str.substr(-4, 2));             // th                    
        console.log(str.substr(6));                 // Suthar                    
        console.log(str.substr(-6));                // Suthar                     
        console.log(str.substr());                  // Karan Suthar                   
        console.log(str.substr(5445));              //                        
        console.log(str.substr(-445));              // Karan Suthar                       
        console.log(str.substr(-31, 65));           // Karan Suthar

toLowerCase()

  1. toLowerCase()
  2. The toLowerCase() method returns the value of the string converted to lower case. toLowerCase() does not affect the value of the string str itself

toUpperCase()

  1. toUpperCase()
  2. The toUpperCase() method returns the value of the string converted to uppercase. This method does not affect the value of the string itself since JavaScript strings are immutable

Example:

        str = "kaRaN sUtHAR"

        console.log(str.toLowerCase()); // karan suthar
        console.log(str.toUpperCase()); // KARAN SUTHAR

trim()

  1. trim()
  2. A new string representing str stripped of whitespace from both its beginning and end. Whitespace is defined as white space characters plus line terminators, If neither the beginning or end of str has any whitespace, a new string is still returned (essentially a copy of str)

trimStart()

  1. trimStart()
  2. A new string representing str stripped of whitespace from its beginning (left side). Whitespace is defined as white space characters plus line terminators, If the beginning of str has no whitespace, a new string is still returned (essentially a copy of str)

trimEnd()

  1. trimEnd()
  2. A new string representing str stripped of whitespace from its end (right side). Whitespace is defined as white space characters plus line terminators, If the end of str has no whitespace, a new string is still returned (essentially a copy of str)

Example:

        txt1 = "    Karan    Suthar    "

        str1 = txt1.trimStart();
        str2 = txt1.trimEnd();
        str3 = txt1.trim();
        
        console.log(str1)  // Karan    Suthar     
        console.log(str2)  //     Karan    Suthar 
        console.log(str3)  // Karan    Suthar

split()

  1. split(separator)
    split(separator, limit)
  2. Returns An Array of strings, split at each point where the separator occurs in the given string.

Example:

        const str = "Karan Suthar, Kese Hain Aap, Sab Badhiya!!!"

        console.log(str.split())           // [ 'Karan Suthar, Kese Hain Aap, Sab Badhiya!!!' ]     
        console.log(str.split(" "))        // [ 'Karan', 'Suthar,', 'Kese', 'Hain', 'Aap,', 'Sab', 'Badhiya!!!' ]       
        console.log(str.split(","))        // [ 'Karan Suthar', ' Kese Hain Aap', ' Sab Badhiya!!!' ]      
        console.log(str.split(",", 1))     // [ 'Karan Suthar' ]           
        console.log(str.split(",", 2))     // [ 'Karan Suthar', ' Kese Hain Aap' ]           
        console.log(str.split("Suthar"));  // [ 'Karan ', ', Kese Hain Aap, Sab Badhiya!!!' ]
        console.log(str.split("&"));       // [ 'Karan Suthar, Kese Hain Aap, Sab Badhiya!!!' ]
