C++ String Methods

String Length

  1. length() or size()
  2. string_object.size() or string_object.length()
  3. This function returns the number of characters in the string object

Example:

        #include <iostream>
        #include <string.h>
        using namespace std;
        
        int main()
        {
            string str = "Karan";
        
            cout << str.size() << endl;       // 5
            cout << str.length() << endl;     // 5
            cout << str.capacity() << endl;   // 15
            cout << sizeof(str) << endl;      // 32
        
            char *newStr = "Karan";
        
            cout << strlen(newStr) << endl;   // 5
            cout << sizeof(newStr) << endl;   // 8        
        }

Accessing Character

  1. at()
  2. string_object.at(index);
  3. This function returns the character present at the index

Example:

        #include <iostream>
        #include <string.h>
        using namespace std;
        
        int main()
        {
            string str = "Karan Suthar";
        
            cout << str.at(0) << endl;       // K
            cout << str.at(1) << endl;       // a
            cout << str.at(2) << endl;       // r
            cout << str.at(3) << endl;       // a
            cout << str.at(4) << endl;       // n
            cout << str.at(5) << endl;       //  
            cout << str.at(6) << endl;       // S
            cout << str.at(7) << endl;       // u
            // cout << str.at(-1) << endl;   // Error: std::out_of_range
            // cout << str.at(100) << endl;  // Error: std::out_of_range

            cout << str[0] << endl;          // K
            cout << str[1] << endl;          // a
            cout << str[2] << endl;          // r
            cout << str[3] << endl;          // a
            cout << str[4] << endl;          // n
            cout << str[5] << endl;          //  
            cout << str[6] << endl;          // S
            cout << str[7] << endl;          // u
            // cout << str[-1] << endl;      // Garbage
            // cout << str[100] << endl;     // Garbage
        }

Concatenating Strings

  1. + operator
  2. string_object1 + string_object2
  3. + operator is used to concatenate two strings
  • or
  1. append()
  2. string_object1.append(string2)
  3. Returns reference to the final string

Example:

        #include <iostream>
        #include <string.h>
        using namespace std;
        
        int main()
        {
            string str1 = "Karan";
            string str2 = "Suthar";
        
            string str3 = str1 + " " + str2;
        
            cout << str1 << endl; // Karan
            cout << str2 << endl; // Suthar
        
            string str4 = (str1.append(" ")).append(str2);
        
            cout << str1 << endl; // Karan Suthar
            cout << str2 << endl; // Suthar
            cout << str3 << endl; // Karan Suthar
            cout << str4 << endl; // Karan Suthar
        }

String Comparison

  1. == operator
  2. string_object1 == string_object2
  3. Returns 1(true) if both the strings are equal, otherwise returns 0(false)
  • or
  1. compare()
  2. str1.compare(str2);
  3. Compares two strings and returns
    • 0, if the strings are equal
    • Positive value, if str1 is greater than str2
    • Negative value, if str1 is less than str2

Example:

        #include <iostream>
        #include <string.h>
        using namespace std;

        int main()
        {
            char *str1 = "Hello";
            char str2[] = "Hello";
            char str3[] = "Hi"; 
            char str4[10] = "Karan";
            char str5[13] = "KaraN";

            string str6 = "Hello";
            string str7 = "Hello";
            string str8 = "Hi";
            string str9 = "Karan";
            string str10 = "KaraN";

            int value = str9.compare(str10);

            printf("%d\n", strcmp(str1, str2));  // 0
            printf("%d\n", strcmp(str1, str3));  // -4
            printf("%d\n", strcmp(str4, str5));  // 32
            printf("%d\n", strcmp(str5, str4));  // -32

            printf("%d\n", str6.compare(str7));  // 0
            printf("%d\n", str6.compare(str8));  // -4
            printf("%d\n", value);               // 32
            printf("%d\n", str10.compare(str9)); // -32

            printf("%d\n", str6 == str7);        // 1
            printf("%d\n", str6 == str8);        // 0
            printf("%d\n", str9 == str10);       // 0
            printf("%d\n", str10 == str9);       // 0
        }

Searching

  1. find()
  2. str1.find(var);
  3. var: It can be a C style string, C++ style string, or a character that is to be searched in the string
  4. It returns the index to the first occurrence of the character or a substring in the string

Example:

        #include <iostream>
        #include <string.h>
        using namespace std;

        int main()
        {
            string str = "Karan Suthar";

            int index = str.find("S");

            cout << index << endl;                  // 6
            cout << str.find("thar") << endl;       // 8
            cout << str.find("ran") << endl;        // 2
            cout << str.find(' ') << endl;          // 5
            cout << str.find("Kese Hain") << endl;  // Garbage
        }

Replacing

  1. replace()
  2. str1.replace(index, size, str2);
  3. Returns reference to the str1

Example:

        #include <iostream>
        #include <string.h>
        using namespace std;

        int main()
        {
            string str = "Karan Suthar";
    
            string newStr = str.replace(str.find("Suthar"), 6, "Sharma");
            // string newStr = str.replace(6, 6, "Sharma");

            cout << str << endl;     // Karan Sharma
            cout << newStr << endl;  // Karan Sharma
        }

Substring

  1. substr()
  2. str1.substr(start, numOfCharacters);
  3. It returns the newly created string object

Example:

        #include <iostream>
        #include <string.h>
        using namespace std;

        int main()
        {
            string str = "Karan Suthar";

            string resultStr1 = str.substr(0, 5); 
            // string resultStr2 = str.substr(-1, 53); // Error: std::out_of_range

            cout << resultStr1 << endl; // Karan
            cout << str << endl;        // Karan Suthar
        }

Inserting

  1. insert()
  2. str1.insert(index, str2);
  3. It returns reference to the str1

Example:

        #include <iostream>
        #include <string.h>
        using namespace std;

        int main()
        {
            string str = "Karan Suthar";
    
            // string newStr = str.insert(50, " Jii"); // Error: std::out_of_range
            string newStr = str.insert(5, " Jii");

            cout << str << endl;     // Karan Jii Suthar
            cout << newStr << endl;  // Karan Jii Suthar
        }
